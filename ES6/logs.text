npm i -D @babel/core @babel/preset-env @babel/cli 
npm i -D @babel/plugin-transform-regenerator
npm i -s @babel/runtime

.babelrc 
{
	"presets": [
	],
	"plugins": [
		["@babel/plugin-transform-regenerator", {
			"asyncGenerators": true,
			"generators": true,
			"async": true
		}],
		"@babel/plugin-transform-runtime"
	]
}
